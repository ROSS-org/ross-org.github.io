509//compares the 'new' event to the signature. If the new event is to occur
510//n_sig later (larger) than e_sig signature, return -1
511//n_sig before (smaller) than e_sig signature, return 1
512//at the signature - return 0
513static inline int tw_event_sig_compare_ptr(tw_event_sig const * e_sig, tw_event_sig const * n_sig)
514{
515 int const time_compare = TW_STIME_CMP(e_sig->recv_ts, n_sig->recv_ts);
516 if (time_compare != 0) {
517 return time_compare;
518 }
519
520 // then we compare the user defined priority first
521 int const prio_compare = TW_STIME_CMP(e_sig->priority, n_sig->priority);
522 if (prio_compare != 0) {
523 return prio_compare;
524 }
525
526 // if tie with user pririty then we use tiebreaker
527 int const min_len = min_int(e_sig->tie_lineage_length, n_sig->tie_lineage_length);
528 for (int i = 0; i < min_len; i++) // lexicographical ordering
529 {
530 if (e_sig->event_tiebreaker[i] < n_sig->event_tiebreaker[i]) {
531 return -1;
532 } else if (e_sig->event_tiebreaker[i] > n_sig->event_tiebreaker[i]) {
533 return 1;
534 }
535 }
536
537 if (e_sig->tie_lineage_length == n_sig->tie_lineage_length) { //total tie
538 return 0;
539 } else if (e_sig->tie_lineage_length > n_sig->tie_lineage_length) {//give priority to one with shorter lineage
540 return 1;
541 } else {
542 return -1;
543 }
544}
