00045 /*
00046  * CPU_HAS_EFFICIENT_UNALIGNED_MEMORY_ACCESS :
00047  * By default, the source code expects the compiler to correctly optimize
00048  * 4-bytes and 8-bytes read on architectures able to handle it efficiently.
00049  * This is not always the case. In some circumstances (ARM notably),
00050  * the compiler will issue cautious code even when target is able to correctly handle unaligned memory accesses.
00051  *
00052  * You can force the compiler to use unaligned memory access by uncommenting the line below.
00053  * One of the below scenarios will happen :
00054  * 1 - Your target CPU correctly handle unaligned access, and was not well optimized by compiler (good case).
00055  *     You will witness large performance improvements (+50% and up).
00056  *     Keep the line uncommented and send a word to upstream (https://groups.google.com/forum/#!forum/lz4c)
00057  *     The goal is to automatically detect such situations by adding your target CPU within an exception list.
00058  * 2 - Your target CPU correctly handle unaligned access, and was already already optimized by compiler
00059  *     No change will be experienced.
00060  * 3 - Your target CPU inefficiently handle unaligned access.
00061  *     You will experience a performance loss. Comment back the line.
00062  * 4 - Your target CPU does not handle unaligned access.
00063  *     Program will crash.
00064  * If uncommenting results in better performance (case 1)
00065  * please report your configuration to upstream (https://groups.google.com/forum/#!forum/lz4c)
00066  * An automatic detection macro will be added to match your case within future versions of the library.
00067  */

00386 static unsigned LZ4_count(const BYTE* pIn, const BYTE* pMatch, const BYTE* pInLimit)
00387 {
00388     const BYTE* const pStart = pIn;
00389 
00390     while (likely(pIn<pInLimit-(STEPSIZE-1)))
00391     {
00392         size_t diff = LZ4_read_ARCH(pMatch) ^ LZ4_read_ARCH(pIn);
00393         if (!diff) { pIn+=STEPSIZE; pMatch+=STEPSIZE; continue; }
00394         pIn += LZ4_NbCommonBytes(diff);
00395         return (unsigned)(pIn - pStart);
00396     }
00397 
00398     if (LZ4_64bits()) if ((pIn<(pInLimit-3)) && (LZ4_read32(pMatch) == LZ4_read32(pIn))) { pIn+=4; pMatch+=4; }
00399     if ((pIn<(pInLimit-1)) && (LZ4_read16(pMatch) == LZ4_read16(pIn))) { pIn+=2; pMatch+=2; }
00400     if ((pIn<pInLimit) && (*pMatch == *pIn)) pIn++;
00401     return (unsigned)(pIn - pStart);
00402 }

00495                  const char* source,
00496                  char* dest,
00497                  int inputSize,
00498                  int maxOutputSize,
00499                  limitedOutput_directive outputLimited,
00500                  tableType_t tableType,
00501                  dict_directive dict,
00502                  dictIssue_directive dictIssue)
00503 {
00504     LZ4_stream_t_internal* const dictPtr = (LZ4_stream_t_internal*)ctx;
00505 
00506     const BYTE* ip = (const BYTE*) source;
00507     const BYTE* base;
00508     const BYTE* lowLimit;
00509     const BYTE* const lowRefLimit = ip - dictPtr->dictSize;
00510     const BYTE* const dictionary = dictPtr->dictionary;
00511     const BYTE* const dictEnd = dictionary + dictPtr->dictSize;
00512     const size_t dictDelta = dictEnd - (const BYTE*)source;
00513     const BYTE* anchor = (const BYTE*) source;
00514     const BYTE* const iend = ip + inputSize;
00515     const BYTE* const mflimit = iend - MFLIMIT;
00516     const BYTE* const matchlimit = iend - LASTLITERALS;
00517 
00518     BYTE* op = (BYTE*) dest;
00519     BYTE* const olimit = op + maxOutputSize;
00520 
00521     const int skipStrength = SKIPSTRENGTH;
00522     U32 forwardH;
00523     size_t refDelta=0;
00524 
00525     /* Init conditions */
00526     if ((U32)inputSize > (U32)LZ4_MAX_INPUT_SIZE) return 0;          /* Unsupported input size, too large (or negative) */
00527     switch(dict)
00528     {
00529     case noDict:
00530     default:
00531         base = (const BYTE*)source;
00532         lowLimit = (const BYTE*)source;
00533         break;
00534     case withPrefix64k:
00535         base = (const BYTE*)source - dictPtr->currentOffset;
00536         lowLimit = (const BYTE*)source - dictPtr->dictSize;
00537         break;
00538     case usingExtDict:
00539         base = (const BYTE*)source - dictPtr->currentOffset;
00540         lowLimit = (const BYTE*)source;
00541         break;
00542     }
00543     if ((tableType == byU16) && (inputSize>=(int)LZ4_64KLIMIT)) return 0;   /* Size too large (not within 64K limit) */
00544     if (inputSize<LZ4_minLength) goto _last_literals;                       /* Input too small, no compression (all literals) */
00545 
00546     /* First Byte */
00547     LZ4_putPosition(ip, ctx, tableType, base);
00548     ip++; forwardH = LZ4_hashPosition(ip, tableType);
00549 
00550     /* Main Loop */
00551     for ( ; ; )
00552     {
00553         const BYTE* match;
00554         BYTE* token;
00555         {
00556             const BYTE* forwardIp = ip;
00557             unsigned step=1;
00558             unsigned searchMatchNb = (1U << skipStrength);
00559 
00560             /* Find a match */
00561             do {
00562                 U32 h = forwardH;
00563                 ip = forwardIp;
00564                 forwardIp += step;
00565                 step = searchMatchNb++ >> skipStrength;
00566 
00567                 if (unlikely(forwardIp > mflimit)) goto _last_literals;
00568 
00569                 match = LZ4_getPositionOnHash(h, ctx, tableType, base);
00570                 if (dict==usingExtDict)
00571                 {
00572                     if (match<(const BYTE*)source)
00573                     {
00574                         refDelta = dictDelta;
00575                         lowLimit = dictionary;
00576                     }
00577                     else
00578                     {
00579                         refDelta = 0;
00580                         lowLimit = (const BYTE*)source;
00581                     }
00582                 }
00583                 forwardH = LZ4_hashPosition(forwardIp, tableType);
00584                 LZ4_putPositionOnHash(ip, h, ctx, tableType, base);
00585 
00586             } while ( ((dictIssue==dictSmall) ? (match < lowRefLimit) : 0)
00587                 || ((tableType==byU16) ? 0 : (match + MAX_DISTANCE < ip))
00588                 || (LZ4_read32(match+refDelta) != LZ4_read32(ip)) );
00589         }
00590 
00591         /* Catch up */
00592         while ((ip>anchor) && (match+refDelta > lowLimit) && (unlikely(ip[-1]==match[refDelta-1]))) { ip--; match--; }
00593 
00594         {
00595             /* Encode Literal length */
00596             unsigned litLength = (unsigned)(ip - anchor);
00597             token = op++;
00598             if ((outputLimited) && (unlikely(op + litLength + (2 + 1 + LASTLITERALS) + (litLength/255) > olimit)))
00599                 return 0;   /* Check output limit */
00600             if (litLength>=RUN_MASK)
00601             {
00602                 int len = (int)litLength-RUN_MASK;
00603                 *token=(RUN_MASK<<ML_BITS);
00604                 for(; len >= 255 ; len-=255) *op++ = 255;
00605                 *op++ = (BYTE)len;
00606             }
00607             else *token = (BYTE)(litLength<<ML_BITS);
00608 
00609             /* Copy Literals */
00610             LZ4_wildCopy(op, anchor, op+litLength);
00611             op+=litLength;
00612         }
00613 
00614 _next_match:
00615         /* Encode Offset */
00616         LZ4_writeLE16(op, (U16)(ip-match)); op+=2;
00617 
00618         /* Encode MatchLength */
00619         {
00620             unsigned matchLength;
00621 
00622             if ((dict==usingExtDict) && (lowLimit==dictionary))
00623             {
00624                 const BYTE* limit;
00625                 match += refDelta;
00626                 limit = ip + (dictEnd-match);
00627                 if (limit > matchlimit) limit = matchlimit;
00628                 matchLength = LZ4_count(ip+MINMATCH, match+MINMATCH, limit);
00629                 ip += MINMATCH + matchLength;
00630                 if (ip==limit)
00631                 {
00632                     unsigned more = LZ4_count(ip, (const BYTE*)source, matchlimit);
00633                     matchLength += more;
00634                     ip += more;
00635                 }
00636             }
00637             else
00638             {
00639                 matchLength = LZ4_count(ip+MINMATCH, match+MINMATCH, matchlimit);
00640                 ip += MINMATCH + matchLength;
00641             }
00642 
00643             if (matchLength>=ML_MASK)
00644             {
00645                 if ((outputLimited) && (unlikely(op + (1 + LASTLITERALS) + (matchLength>>8) > olimit)))
00646                     return 0;    /* Check output limit */
00647                 *token += ML_MASK;
00648                 matchLength -= ML_MASK;
00649                 for (; matchLength >= 510 ; matchLength-=510) { *op++ = 255; *op++ = 255; }
00650                 if (matchLength >= 255) { matchLength-=255; *op++ = 255; }
00651                 *op++ = (BYTE)matchLength;
00652             }
00653             else *token += (BYTE)(matchLength);
00654         }
00655 
00656         anchor = ip;
00657 
00658         /* Test end of chunk */
00659         if (ip > mflimit) break;
00660 
00661         /* Fill table */
00662         LZ4_putPosition(ip-2, ctx, tableType, base);
00663 
00664         /* Test next position */
00665         match = LZ4_getPosition(ip, ctx, tableType, base);
00666         if (dict==usingExtDict)
00667         {
00668             if (match<(const BYTE*)source)
00669             {
00670                 refDelta = dictDelta;
00671                 lowLimit = dictionary;
00672             }
00673             else
00674             {
00675                 refDelta = 0;
00676                 lowLimit = (const BYTE*)source;
00677             }
00678         }
00679         LZ4_putPosition(ip, ctx, tableType, base);
00680         if ( ((dictIssue==dictSmall) ? (match>=lowRefLimit) : 1)
00681             && (match+MAX_DISTANCE>=ip)
00682             && (LZ4_read32(match+refDelta)==LZ4_read32(ip)) )
00683         { token=op++; *token=0; goto _next_match; }
00684 
00685         /* Prepare next loop */
00686         forwardH = LZ4_hashPosition(++ip, tableType);
00687     }
00688 
00689 _last_literals:
00690     /* Encode Last Literals */
00691     {
00692         int lastRun = (int)(iend - anchor);
00693         if ((outputLimited) && (((char*)op - dest) + lastRun + 1 + ((lastRun+255-RUN_MASK)/255) > (U32)maxOutputSize))
00694             return 0;   /* Check output limit */
00695         if (lastRun>=(int)RUN_MASK) { *op++=(RUN_MASK<<ML_BITS); lastRun-=RUN_MASK; for(; lastRun >= 255 ; lastRun-=255) *op++ = 255; *op++ = (BYTE) lastRun; }
00696         else *op++ = (BYTE)(lastRun<<ML_BITS);
00697         memcpy(op, anchor, iend - anchor);
00698         op += iend-anchor;
00699     }
00700 
00701     /* End */
00702     return (int) (((char*)op)-dest);
00703 }

00808 static void LZ4_renormDictT(LZ4_stream_t_internal* LZ4_dict, const BYTE* src)
00809 {
00810     if ((LZ4_dict->currentOffset > 0x80000000) ||
00811         ((size_t)LZ4_dict->currentOffset > (size_t)src))   /* address space overflow */
00812     {
00813         /* rescale hash table */
00814         U32 delta = LZ4_dict->currentOffset - 64 KB;
00815         const BYTE* dictEnd = LZ4_dict->dictionary + LZ4_dict->dictSize;
00816         int i;
00817         for (i=0; i<HASH_SIZE_U32; i++)
00818         {
00819             if (LZ4_dict->hashTable[i] < delta) LZ4_dict->hashTable[i]=0;
00820             else LZ4_dict->hashTable[i] -= delta;
00821         }
00822         LZ4_dict->currentOffset = 64 KB;
00823         if (LZ4_dict->dictSize > 64 KB) LZ4_dict->dictSize = 64 KB;
00824         LZ4_dict->dictionary = dictEnd - LZ4_dict->dictSize;
00825     }

00929 /****************************
00930    Decompression functions
00931 ****************************/
00932 /*
00933  * This generic decompression function cover all use cases.
00934  * It shall be instantiated several times, using different sets of directives
00935  * Note that it is essential this generic function is really inlined,
00936  * in order to remove useless branches during compilation optimization.
00937  */
00938 FORCE_INLINE int LZ4_decompress_generic(
00939                  const char* const source,
00940                  char* const dest,
00941                  int inputSize,
00942                  int outputSize,         /* If endOnInput==endOnInputSize, this value is the max size of Output Buffer. */
00943 
00944                  int endOnInput,         /* endOnOutputSize, endOnInputSize */
00945                  int partialDecoding,    /* full, partial */
00946                  int targetOutputSize,   /* only used if partialDecoding==partial */
00947                  int dict,               /* noDict, withPrefix64k, usingExtDict */
00948                  const BYTE* const lowPrefix,  /* == dest if dict == noDict */
00949                  const BYTE* const dictStart,  /* only if dict==usingExtDict */
00950                  const size_t dictSize         /* note : = 0 if noDict */
00951                  )
00952 {
00953     /* Local Variables */
00954     const BYTE* restrict ip = (const BYTE*) source;
00955     const BYTE* const iend = ip + inputSize;
00956 
00957     BYTE* op = (BYTE*) dest;
00958     BYTE* const oend = op + outputSize;
00959     BYTE* cpy;
00960     BYTE* oexit = op + targetOutputSize;
00961     const BYTE* const lowLimit = lowPrefix - dictSize;
00962 
00963     const BYTE* const dictEnd = (const BYTE*)dictStart + dictSize;
00964     const size_t dec32table[] = {4, 1, 2, 1, 4, 4, 4, 4};
00965     const size_t dec64table[] = {0, 0, 0, (size_t)-1, 0, 1, 2, 3};
00966 
00967     const int safeDecode = (endOnInput==endOnInputSize);
00968     const int checkOffset = ((safeDecode) && (dictSize < (int)(64 KB)));
00969 
00970 
00971     /* Special cases */
00972     if ((partialDecoding) && (oexit> oend-MFLIMIT)) oexit = oend-MFLIMIT;                         /* targetOutputSize too high => decode everything */
00973     if ((endOnInput) && (unlikely(outputSize==0))) return ((inputSize==1) && (*ip==0)) ? 0 : -1;  /* Empty output buffer */
00974     if ((!endOnInput) && (unlikely(outputSize==0))) return (*ip==0?1:-1);
00975 
00976 
00977     /* Main Loop */
00978     while (1)
00979     {
00980         unsigned token;
00981         size_t length;
00982         const BYTE* match;
00983 
00984         /* get literal length */
00985         token = *ip++;
00986         if ((length=(token>>ML_BITS)) == RUN_MASK)
00987         {
00988             unsigned s;
00989             do
00990             {
00991                 s = *ip++;
00992                 length += s;
00993             }
00994             while (likely((endOnInput)?ip<iend-RUN_MASK:1) && (s==255));
00995             if ((safeDecode) && unlikely((size_t)(op+length)<(size_t)(op))) goto _output_error;   /* overflow detection */
00996             if ((safeDecode) && unlikely((size_t)(ip+length)<(size_t)(ip))) goto _output_error;   /* overflow detection */
00997         }
00998 
00999         /* copy literals */
01000         cpy = op+length;
01001         if (((endOnInput) && ((cpy>(partialDecoding?oexit:oend-MFLIMIT)) || (ip+length>iend-(2+1+LASTLITERALS))) )
01002             || ((!endOnInput) && (cpy>oend-COPYLENGTH)))
01003         {
01004             if (partialDecoding)
01005             {
01006                 if (cpy > oend) goto _output_error;                           /* Error : write attempt beyond end of output buffer */
01007                 if ((endOnInput) && (ip+length > iend)) goto _output_error;   /* Error : read attempt beyond end of input buffer */
01008             }
01009             else
01010             {
01011                 if ((!endOnInput) && (cpy != oend)) goto _output_error;       /* Error : block decoding must stop exactly there */
01012                 if ((endOnInput) && ((ip+length != iend) || (cpy > oend))) goto _output_error;   /* Error : input must be consumed */
01013             }
01014             memcpy(op, ip, length);
01015             ip += length;
01016             op += length;
01017             break;     /* Necessarily EOF, due to parsing restrictions */
01018         }
01019         LZ4_wildCopy(op, ip, cpy);
01020         ip += length; op = cpy;
01021 
01022         /* get offset */
01023         match = cpy - LZ4_readLE16(ip); ip+=2;
01024         if ((checkOffset) && (unlikely(match < lowLimit))) goto _output_error;   /* Error : offset outside destination buffer */
01025 
01026         /* get matchlength */
01027         length = token & ML_MASK;
01028         if (length == ML_MASK)
01029         {
01030             unsigned s;
01031             do
01032             {
01033                 if ((endOnInput) && (ip > iend-LASTLITERALS)) goto _output_error;
01034                 s = *ip++;
01035                 length += s;
01036             } while (s==255);
01037             if ((safeDecode) && unlikely((size_t)(op+length)<(size_t)op)) goto _output_error;   /* overflow detection */
01038         }
01039         length += MINMATCH;
01040 
01041         /* check external dictionary */
01042         if ((dict==usingExtDict) && (match < lowPrefix))
01043         {
01044             if (unlikely(op+length > oend-LASTLITERALS)) goto _output_error;   /* doesn't respect parsing restriction */
01045 
01046             if (length <= (size_t)(lowPrefix-match))
01047             {
01048                 /* match can be copied as a single segment from external dictionary */
01049                 match = dictEnd - (lowPrefix-match);
01050                 memcpy(op, match, length);
01051                 op += length;
01052             }
01053             else
01054             {
01055                 /* match encompass external dictionary and current segment */
01056                 size_t copySize = (size_t)(lowPrefix-match);
01057                 memcpy(op, dictEnd - copySize, copySize);
01058                 op += copySize;
01059                 copySize = length - copySize;
01060                 if (copySize > (size_t)(op-lowPrefix))   /* overlap within current segment */
01061                 {
01062                     BYTE* const endOfMatch = op + copySize;
01063                     const BYTE* copyFrom = lowPrefix;
01064                     while (op < endOfMatch) *op++ = *copyFrom++;
01065                 }
01066                 else
01067                 {
01068                     memcpy(op, lowPrefix, copySize);
01069                     op += copySize;
01070                 }
01071             }
01072             continue;
01073         }
01074 
01075         /* copy repeated sequence */
01076         cpy = op + length;
01077         if (unlikely((op-match)<8))
01078         {
01079             const size_t dec64 = dec64table[op-match];
01080             op[0] = match[0];
01081             op[1] = match[1];
01082             op[2] = match[2];
01083             op[3] = match[3];
01084             match += dec32table[op-match];
01085             LZ4_copy4(op+4, match);
01086             op += 8; match -= dec64;
01087         } else { LZ4_copy8(op, match); op+=8; match+=8; }
01088 
01089         if (unlikely(cpy>oend-12))
01090         {
01091             if (cpy > oend-LASTLITERALS) goto _output_error;    /* Error : last LASTLITERALS bytes must be literals */
01092             if (op < oend-8)
01093             {
01094                 LZ4_wildCopy(op, match, oend-8);
01095                 match += (oend-8) - op;
01096                 op = oend-8;
01097             }
01098             while (op<cpy) *op++ = *match++;
01099         }
01100         else
01101             LZ4_wildCopy(op, match, cpy);
01102         op=cpy;   /* correction */
01103     }
01104 
01105     /* end of decoding */
01106     if (endOnInput)
01107        return (int) (((char*)op)-dest);     /* Nb of output bytes decoded */
01108     else
01109        return (int) (((char*)ip)-source);   /* Nb of input bytes read */
01110 
01111     /* Overflow error detected */
01112 _output_error:
01113     return (int) (-(((char*)ip)-source))-1;
01114 }
