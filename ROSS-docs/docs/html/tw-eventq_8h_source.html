00114   /* Nothing to collect from this event list? */
00115   if (!t || t->recv_ts >= gvt)
00116     return;
00117 
00118   if (h->recv_ts < gvt)
00119     {
00120       /* Everything in the queue can be collected */
00121       tw_eventq_push_list(&pe->free_q, h, t, q->size);

00124     } else {
00125     /* Only some of the list can be collected.  We'll wind up

00129      * these conditions.
00130      */

00132 
00133     /* Search the leading part of the list... */
00134     for (h = t->prev, cnt = 1; h && h->recv_ts < gvt; cnt++)
00135       h = h->prev;
00136

00138     n = h;

00140     /* Back up one cell, we overshot where to cut the list */
00141     h = h->next;
