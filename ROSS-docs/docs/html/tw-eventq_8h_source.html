118 static inline void
119 tw_eventq_fossil_collect(tw_eventq *q, tw_pe *pe)
120 {
121  tw_stime gvt = pe->GVT;
122 
123  tw_event *h = q->head;
124  tw_event *t = q->tail;
125 
126  int cnt;
127 
128  /* Nothing to collect from this event list? */
129  if (!t || t->recv_ts >= gvt)
130  return;
131 
132  if (h->recv_ts < gvt)
133  {
134  /* Everything in the queue can be collected */
135  tw_eventq_push_list(&pe->free_q, h, t, q->size);
136  q->head = q->tail = NULL;
137  q->size = 0;
138  } else {
139  /* Only some of the list can be collected. We'll wind up
140  * with at least one event being collected and at least
141  * another event staying behind in the eventq structure so
142  * we can really optimize this list splicing operation for
143  * these conditions.
144  */
145  tw_event *n;
146 
147  /* Search the leading part of the list... */
148  for (h = t->prev, cnt = 1; h && h->recv_ts < gvt; cnt++)
149  h = h->prev;
150 
151  /* t isn't eligible for collection; its the new head */
152  n = h;
153 
154  /* Back up one cell, we overshot where to cut the list */
155  h = h->next;
156 
157  /* Cut h..t out of the event queue */
158  q->tail = n;
159  n->next = NULL;
160  q->size -= cnt;
161 
162  /* Free h..t (inclusive) */
163  tw_eventq_push_list(&pe->free_q, h, t, cnt);
164  }
165 }

168 tw_eventq_alloc(tw_eventq * q, unsigned int cnt)
169 {
170  tw_event *event;
171  size_t event_len;
172  size_t align;
173 #ifdef ROSS_ALLOC_DEBUG
174  tw_event *event_prev = NULL;
175 #endif
176 
177  /* Construct a linked list of free events. We allocate
178  * the events such that they look like this in memory:
179  *
180  * ------------------
181  * | tw_event |
182  * | user_data |
183  * ------------------
184  * | tw_event |
185  * | user_data |
186  * ------------------
187  * ......
188  * ------------------
189  */
190 
191  align = ROSS_MAX(sizeof(double), sizeof(void*));
192  event_len = sizeof(tw_event) + g_tw_msg_sz;
193  if (event_len & (align - 1))
194  {
195  event_len += align - (event_len & (align - 1));
196  //tw_error(TW_LOC, "REALIGNING EVENT MEMORY!\n");
197  }
198 #if ROSS_MEMORY
199  if(event_len > TW_MEMORY_BUFFER_SIZE)
200  {
201  tw_error(TW_LOC, "g_tw_msg_sz of %d is too large.\nThe ROSS_MEMORY maximum buffer size (including space reserved for alignment and header) is %d.\nPlease reduce the message size specified by the model using tw_define_lps().", g_tw_msg_sz, TW_MEMORY_BUFFER_SIZE);
202  }
203 #endif
204  g_tw_event_msg_sz = event_len;
205 
206  // compute number of events needed for the network.
207  g_tw_gvt_threshold = (int) ceil(g_tw_net_device_size / g_tw_event_msg_sz);
208  g_tw_gvt_threshold = g_tw_net_device_size;
209  g_tw_events_per_pe += g_tw_gvt_threshold;
210  cnt += g_tw_gvt_threshold;
211 
212  q->size = cnt;
213  /* allocate one at a time so tools like valgrind can detect buffer
214  * overflows */
215 #ifdef ROSS_ALLOC_DEBUG
216  q->head = event = (tw_event *)tw_calloc(TW_LOC, "events", event_len, 1);
217  while (--cnt) {
218  event->state.owner = TW_pe_free_q;
219  event->prev = event_prev;
220  event_prev = event;
221  event->next = (tw_event *) tw_calloc(TW_LOC, "events", event_len, 1);
222  event = event->next;
223  }
224  event->prev = event_prev;
225 #else
226  /* alloc in one large block for performance/locality */
227  q->head = event = (tw_event *)tw_calloc(TW_LOC, "events", event_len, cnt);
228  while (--cnt) {
229  event->state.owner = TW_pe_free_q;
230  event->prev = (tw_event *) (((char *)event) - event_len);
231  event->next = (tw_event *) (((char *)event) + event_len);
232  event = event->next;
233  }
234  event->prev = (tw_event *) (((char *)event) - event_len);
235 #endif
236 
237  event->state.owner = TW_pe_free_q;
238  q->head->prev = event->next = NULL;
239  q->tail = event;
240 }
