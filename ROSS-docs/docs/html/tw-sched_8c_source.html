3 /**
4  * \brief Reset the event bitfield prior to entering the event handler
5  * post-reverse - reset the bitfield so that a potential re-running of the
6  * event is presented with a consistent bitfield state
7  * NOTE: the size checks are to better support the experimental reverse
8  * computation compiler, which can use a larger bitfield.
9  * Courtesy of John P Jenkins
10  */
11 static inline void reset_bitfields(tw_event *revent)
12 {
13  if (sizeof(revent->cv) == sizeof(uint32_t)){
14  *(uint32_t*)&revent->cv = 0;
15  }
16  else if (sizeof(revent->cv) == sizeof(uint64_t)){
17  *(uint64_t*)&revent->cv = 0;
18  }
19  else{
20  memset(&revent->cv, 0, sizeof(revent->cv));
21  }
22 }

36  cev = tw_eventq_pop_list(&me->event_q);
37 
38  for (; cev; cev = nev) {
39  nev = cev->next;
40 
41  if(!cev->state.owner || cev->state.owner == TW_pe_free_q) {
42  tw_error(TW_LOC, "no owner!");
43  }
44  if (cev->state.cancel_q) {
45  cev->state.owner = TW_pe_anti_msg;
46  cev->next = cev->prev = NULL;
47  continue;
48  }
49

51  case TW_pe_event_q:
52  dest_kp = cev->dest_lp->kp;
53

56  * after we processed events occuring after it.
57  * We need to jump back to before cev's timestamp.
58  */
59  start = tw_clock_read();
60  tw_kp_rollback_to(dest_kp, cev->recv_ts);
61  me->stats.s_rollback += tw_clock_read() - start;
62  if (g_st_ev_trace == RB_TRACE)
63  st_collect_event_data(cev, (double)start / g_tw_clock_rate, 0);
64  }
65  start = tw_clock_read();
66  tw_pq_enqueue(me->pq, cev);
67  me->stats.s_pq += tw_clock_read() - start;
68  break;
69 
70  default:
71  tw_error(TW_LOC, "Event in event_q, but owner %d not recognized", cev->state.owner);
72  }
73  }
74  }

82 static void tw_sched_cancel_q(tw_pe * me) {
83  tw_clock start;
84  tw_event *cev;
85  tw_event *nev;
86 
87  start = tw_clock_read();

89  cev = me->cancel_q;
90  me->cancel_q = NULL;
91 
92  for (; cev; cev = nev) {
93  nev = cev->cancel_next;
94 
95  if (!cev->state.cancel_q) {
96  tw_error(TW_LOC, "No cancel_q bit on event in cancel_q");
97  }
98 
99  if(!cev->state.owner || cev->state.owner == TW_pe_free_q) {
100  tw_error(TW_LOC, "Cancelled event, no owner!");
101  }
102 
103  switch (cev->state.owner) {
104  case TW_pe_event_q:
105  /* This event hasn't been added to our pq yet and we
106  * have not officially received it yet either. We'll
107  * do the actual free of this event when we receive it
108  * as we spin out the event_q chain.
109  */
110  tw_eventq_delete_any(&me->event_q, cev);
111 
112  tw_event_free(me, cev);
113  break;
114 
115  case TW_pe_anti_msg:
116  tw_event_free(me, cev);
117  break;
118 
119  case TW_pe_pq:
120  /* Event was not cancelled directly from the event_q
121  * because the cancel message came after we popped it
122  * out of that queue but before we could process it.
123  */
124  tw_pq_delete_any(me->pq, cev);
125  tw_event_free(me, cev);
126  break;
127 
128  case TW_kp_pevent_q:
129  /* The event was already processed.
130  * SECONDARY ROLLBACK
131  */
132  tw_kp_rollback_event(cev);
133  tw_event_free(me, cev);
134  break;
135 
136  default:
137  tw_error(TW_LOC, "Event in cancel_q, but owner %d not recognized", cev->state.owner);
138  }
139  }
140  }
141 
142  me->stats.s_cancel_q += tw_clock_read() - start;
143 }

728  printf("This schedule assumes the following: \n");
729  printf(" 1) One 1 Processor/Core is used.\n");
730  printf(" 2) One 1 KP is used.\n");
731  printf(" NOTE: use the --nkp=1 argument to the simulation to ensure that\n");
732  printf(" it only uses 1 KP.\n");
733  printf(" 3) Events ARE NEVER RECLAIMED (LP Commit Functions are not called).\n");
734  printf(" 4) Executes til out of memory (16 events left) and \n injects rollback to first before primodal init event.\n");
735  printf(" 5) g_tw_rollback_time = %13.12lf \n", g_tw_rollback_time);
736  printf("/***************************************************************************/\n");
737 
738  if( g_tw_nkp > 1 ) {
739  tw_error(TW_LOC, "Number of KPs is greater than 1.");
740  }
741 
742  tw_wall_now(&me->start_time);
743 
744  while ((cev = tw_pq_dequeue(me->pq))) {
745  tw_lp *clp = cev->dest_lp;
746  tw_kp *ckp = clp->kp;
747 
748  me->cur_event = cev;
749  ckp->last_time = cev->recv_ts;
750 
751  /* don't update GVT */
752  reset_bitfields(cev);
753 
754  // state-save and update the LP's critical path
755  unsigned int prev_cp = clp->critical_path;
756  clp->critical_path = ROSS_MAX(clp->critical_path, cev->critical_path)+1;
757  (*clp->type->event)(clp->cur_state, &cev->cv, tw_event_data(cev), clp);

768  break;
769  }
770  }
