17  {
18  cev = tw_eventq_pop_list(&me->event_q);
19 
20  for (; cev; cev = nev)
21  {
22  nev = cev->next;
23 
24  if(!cev->state.owner || cev->state.owner == TW_pe_free_q)
25  tw_error(TW_LOC, "no owner!");
26 
27  if (cev->state.cancel_q)
28  {

30  cev->next = cev->prev = NULL;

32  }

36  case TW_pe_event_q:

38

40  {

42  * after we processed events occuring after it.
43  * We need to jump back to before cev's timestamp.
44  */
45  start = tw_clock_read();
46  tw_kp_rollback_to(dest_kp, cev->recv_ts);
47  me->stats.s_rollback += tw_clock_read() - start;
48  }
49  start = tw_clock_read();

51  me->stats.s_pq += tw_clock_read() - start;
52  break;
53 
54  default:
55  tw_error(

60  }

123  default:

136 tw_sched_batch(tw_pe * me)
137 {
138  /* Number of consecutive times we gave up because there were no free event buffers. */
139  static int no_free_event_buffers = 0;

146  * (whichever comes first).
147  */

149  tw_event *cev;
150  tw_lp *clp;
151  tw_kp *ckp;
152 
153  /* OUT OF FREE EVENT BUFFERS. BAD.
154  * Go do fossil collect immediately.
155  */

164  }

166

169  break;

171  if(cev->recv_ts == tw_pq_minimum(me->pq))
172  me->stats.s_pe_event_ties++;
173 
174  clp = cev->dest_lp;
175  ckp = clp->kp;
176  me->cur_event = cev;
177  ckp->last_time = cev->recv_ts;
178 
179  /* Save state if no reverse computation is available */
180  if (!clp->type.revent)
181  tw_error(TW_LOC, "Reverse Computation must be implemented!");
182 
183  start = tw_clock_read();
184  (*clp->type.event)(
185  clp->cur_state,
186  &cev->cv,
187  tw_event_data(cev),
188  clp);
189  ckp->s_nevent_processed++;
190  me->stats.s_event_process += tw_clock_read() - start;
191 
192  /* We ran out of events while processing this event. We
193  * cannot continue without doing GVT and fossil collect.
194  */
195 
196  if (me->cev_abort) {
197  start = tw_clock_read();
198  me->stats.s_nevent_abort++;
199  me->cev_abort = 0;
200 
201  tw_event_rollback(cev);
202  tw_pq_enqueue(me->pq, cev);
203 
204  cev = tw_eventq_peek(&ckp->pevent_q);
205  ckp->last_time = cev ? cev->recv_ts : me->GVT;
206 
207  tw_gvt_force_update(me);
208 
209  me->stats.s_event_abort += tw_clock_read() - start;
210 
211  break;
212  }
213 
214  /* Thread current event into processed queue of kp */
215  cev->state.owner = TW_kp_pevent_q;
216  tw_eventq_unshift(&ckp->pevent_q, cev);
217  }
