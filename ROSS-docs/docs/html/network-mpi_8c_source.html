714 void
715 tw_net_cancel(tw_event *e)
716 {
717  tw_pe *src_pe = e->src_lp->pe;
718 
719  switch (e->state.owner) {
720  case TW_net_outq:
721  /* Cancelled before we could transmit it. Do not
722  * transmit the event and instead just release the
723  * buffer back into our own free list.
724  */
725  tw_eventq_delete_any(&outq, e);
726  tw_event_free(src_pe, e);
727 
728  return;
729 
730  break;
731 
732  case TW_net_asend:
733  /* Too late. We've already let MPI start to send
734  * this event over the network. We can't pull it
735  * back now without sending another message to do
736  * the cancel.
737  *
738  * Setting the cancel_q flag will signal us to do
739  * another message send once the current send of
740  * this message is completed.
741  */
742  e->state.cancel_asend = 1;
743  break;
744 
745  case TW_pe_sevent_q:
746  /* Way late; the event was already sent and is in
747  * our sent event queue. Mark it as a cancel and
748  * place it at the front of the outq.
749  */
750  e->state.cancel_q = 1;
751  tw_eventq_unshift(&outq, e);
752  break;
