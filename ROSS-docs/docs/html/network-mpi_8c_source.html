681 void
682 tw_net_cancel(tw_event *e)
683 {
684  tw_pe *src_pe = e->src_lp->pe;
685 
686  switch (e->state.owner) {
687  case TW_net_outq:
688  /* Cancelled before we could transmit it. Do not
689  * transmit the event and instead just release the
690  * buffer back into our own free list.
691  */
692  tw_eventq_delete_any(&outq, e);
693  tw_event_free(src_pe, e);
694 
695  return;
696 
697  break;
698 
699  case TW_net_asend:
700  /* Too late. We've already let MPI start to send
701  * this event over the network. We can't pull it
702  * back now without sending another message to do
703  * the cancel.
704  *
705  * Setting the cancel_q flag will signal us to do
706  * another message send once the current send of
707  * this message is completed.
708  */
709  e->state.cancel_asend = 1;
710  break;
711 
712  case TW_pe_sevent_q:
713  /* Way late; the event was already sent and is in
714  * our sent event queue. Mark it as a cancel and
715  * place it at the front of the outq.
716  */
717  e->state.cancel_q = 1;
718  tw_eventq_unshift(&outq, e);
719  break;
720 
721  default:
722  /* Huh? Where did you come from? Why are we being
723  * told about you? We did not send you so we cannot
724  * cancel you!
725  */
726  tw_error(
727  TW_LOC,
728  "Don't know how to cancel event owned by %u",
729  e->state.owner);
730  }
731 
732  service_queues(src_pe);
