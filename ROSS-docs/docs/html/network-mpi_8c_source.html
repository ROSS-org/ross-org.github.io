557 send_finish(tw_pe *me, tw_event *e, char * buffer)
558 {
559  me->stats.s_nsend_network++;

561  if (e->state.owner == TW_net_asend) {
562  if (e->state.cancel_asend) {
563  /* Event was cancelled during transmission. We must
564  * send another message to pass the cancel flag to
565  * the other node.

647 void
648 tw_net_cancel(tw_event *e)
649 {
650  tw_pe *src_pe = e->src_lp->pe;
651 
652  switch (e->state.owner) {
653  case TW_net_outq:
654  /* Cancelled before we could transmit it. Do not
655  * transmit the event and instead just release the
656  * buffer back into our own free list.
657  */
658  tw_eventq_delete_any(&outq, e);
659  tw_event_free(src_pe, e);
660 
661  return;
662 
663  break;
664 
665  case TW_net_asend:
666  /* Too late. We've already let MPI start to send
667  * this event over the network. We can't pull it
668  * back now without sending another message to do
669  * the cancel.
670  *
671  * Setting the cancel_q flag will signal us to do
672  * another message send once the current send of
673  * this message is completed.
674  */
675  e->state.cancel_asend = 1;
676  break;
677 
678  case TW_pe_sevent_q:
679  /* Way late; the event was already sent and is in
680  * our sent event queue. Mark it as a cancel and
681  * place it at the front of the outq.
682  */
683  e->state.cancel_q = 1;
684  tw_eventq_unshift(&outq, e);
685  break;
686 
687  default:
688  /* Huh? Where did you come from? Why are we being
689  * told about you? We did not send you so we cannot
690  * cancel you!
691  */
692  tw_error(
693  TW_LOC,
694  "Don't know how to cancel event owned by %u",
695  e->state.owner);
696  }
697 
698  service_queues(src_pe);
699 }
